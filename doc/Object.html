def bubble_sort!(array)
  loop do
    swapped = false
    1.upto(array.size - 1) do |index|
      if block_given?
        next if yield(array[index - 1], array[index])
      else
        next if array[index - 1] <= array[index]
      end
      array[index - 1], array[index] = array[index], array[index - 1]
      swapped = true
    end
    break unless swapped
  end
  nil
end

def divisors(int)
  divs = [1, int]
  return divs.uniq if int == 1
  denom = 2
  until divs.include?(denom)
    inverse = int / denom
    if int % denom == 0
      divs.push(denom, inverse)
    end
    denom += 1
  end
  divs.uniq.sort
end

def rot13(full_name)
  fixed = full_name.chars.map do |char|
    forward = (char.ord + 13).chr
    backward = (char.ord - 13).chr
    case char
    when 'a'..'m', 'A'..'M' then forward
    when 'n'..'z', 'N'..'Z' then backward
    else                         char
    end
  end
  fixed.join
end

def word_finder(letters, size=letters.size, places=[], inserts=[])
  possibilities = letters.permutation(size).map(&:join)
  possibilities.map { |word| word.insert(places[0], inserts[0]) } if places[0]
  possibilities.map { |word| word.insert(places[1], inserts[1]) } if places[1]
  possibilities
end
